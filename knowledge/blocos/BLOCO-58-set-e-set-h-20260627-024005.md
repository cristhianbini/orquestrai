# BLOCO-58 - set +e; set +H

## Script
~~~bash
set +e; set +H
REPORT=/tmp/nginx_snippets_perms_$(date +%s).txt
MARKER="MK_NGINX_SNIPPETS_PERMS_AUDIT"
grep -q "$MARKER" "$REPORT" 2>/dev/null && exit 0

echo "===== BLOCO-NGINX-SNIPPETS-PERMS-AUDIT =====" | tee $REPORT
echo "Timestamp: $(date '+%Y-%m-%d %H:%M:%S')" | tee -a $REPORT
echo "$MARKER" >> $REPORT
echo "" | tee -a $REPORT

echo "[1] Permissões detalhadas do diretório /etc/nginx/snippets/" | tee -a $REPORT
ls -lhA /etc/nginx/snippets/ 2>&1 | tee -a $REPORT
echo "" | tee -a $REPORT

echo "[2] Busca client_max_body_size em todos .conf de /etc/nginx/" | tee -a $REPORT
find /etc/nginx/ -type f -name "*.conf" -exec echo ">>> {}" \; -exec grep -Hn "client_max_body_size" {} \; 2>/dev/null | tee -a $REPORT
echo "" | tee -a $REPORT

echo "[3] Verificação de valores proxy_request_buffering e proxy_buffers" | tee -a $REPORT
find /etc/nginx/ -type f \( -name "*.conf" -o -name "*.include" \) -exec grep -Hn -E "(proxy_request_buffering|proxy_buffers)" {} \; 2>/dev/null | tee -a $REPORT
echo "" | tee -a $REPORT

echo "[4] Existência dos diretórios include críticos" | tee -a $REPORT
for DIR in /etc/nginx/conf.d /etc/nginx/sites-enabled /etc/nginx/snippets; do
    if [ -d "$DIR" ]; then
        echo "  [OK] $DIR existe" | tee -a $REPORT
        echo "       Quantidade de arquivos: $(find $DIR -type f | wc -l)" | tee -a $REPORT
    else
        echo "  [ALERTA] $DIR não encontrado" | tee -a $REPORT
    fi
done
echo "" | tee -a $REPORT

echo "[5] Últimas 30 linhas de /var/log/nginx/error.log" | tee -a $REPORT
if [ -f /var/log/nginx/error.log ]; then
    tail -n 30 /var/log/nginx/error.log | tee -a $REPORT
else
    echo "  -> error.log não encontrado" | tee -a $REPORT
fi
echo "" | tee -a $REPORT

echo "[6] Sintaxe nginx confirmada (nginx -t)" | tee -a $REPORT
nginx -t 2>&1 | tee -a $REPORT
echo "" | tee -a $REPORT

echo "Relatório salvo em: $REPORT" | tee -a $REPORT
echo "===== FIM BLOCO-NGINX-SNIPPETS-PERMS-AUDIT ====="
~~~
