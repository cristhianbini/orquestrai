# BLOCO-134 - set +e; set +H

## Script
~~~bash
set +e; set +H
if [ -d "/var/www/orquestrai/knowledge/" ]; then
    echo "--- Estrutura principal ---"
    ls -la /var/www/orquestrai/knowledge/
    echo "--- Subdiretorios (Metas, Pendencias, etc) ---"
    find /var/www/orquestrai/knowledge/ -maxdepth 1 -type d
    echo "--- Arquivos .md encontrados ---"
    find /var/www/orquestrai/knowledge/ -type f -name "*.md"
else
    echo "Diretorio de conhecimento nao encontrado em /var/www/orquestrai/knowledge/"
fi
~~~
