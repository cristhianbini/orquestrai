# BLOCO-103 - set +e

## Script
~~~bash
set +e
set +H

echo "=== DIAGNÓSTICO ORQUESTRAI STACK ==="
echo ""

echo "1. STATUS DOS CONTAINERS:"
docker ps -a --filter "name=orquestrai" --format "table {{.Names}}\t{{.Status}}\t{{.Ports}}"
echo ""

echo "2. LOGS API (últimas 20 linhas):"
docker logs orquestrai-api --tail 20 2>&1
echo ""

echo "3. HEALTH DA API (interno container):"
docker exec orquestrai-api curl -sf http://localhost:3000/health || echo "❌ API não responde no container"
echo ""

echo "4. NETWORK STACK:"
docker exec orquestrai-api netstat -tlnp 2>/dev/null | grep -E "3000|LISTEN" || echo "netstat não disponível, tentando ss..."
docker exec orquestrai-api ss -tlnp 2>/dev/null | grep -E "3000|LISTEN" || echo "Porta 3000 não está em LISTEN"
echo ""

echo "5. PROCESSO NODE NO CONTAINER:"
docker exec orquestrai-api ps aux | grep -E "node|npm" | grep -v grep
echo ""

echo "6. COMPOSE STATUS:"
cd /var/www/orquestrai && docker compose ps
echo ""

echo "7. ÚLTIMA LICÃO REGISTRADA:"
ls -lht /var/www/orquestrai/knowledge/licoes/ | head -3
~~~
