TEST_BE=/var/www/xmonex_teste/backend

echo "--- Release backend ---" | tee -a $REPORT

echo "" | tee -a $REPORT

  echo "--- $FILE ---" | tee -a $REPORT
  if [ -f "$PROD_BE/$FILE" ]; then
    echo "PROD: $(sha256sum $PROD_BE/$FILE | awk '{print $1}')" | tee -a $REPORT
  else
    echo "PROD: ARQUIVO NÃO ENCONTRADO" | tee -a $REPORT
  fi
  if [ -f "$REL_BE/$FILE" ]; then
    echo "REL:  $(sha256sum $REL_BE/$FILE | awk '{print $1}')" | tee -a $REPORT
  else
    echo "REL:  ARQUIVO NÃO ENCONTRADO" | tee -a $REPORT
  fi
  echo "" | tee -a $REPORT
done

if [ -f "$PROD_BE/.env" ] && [ -f "$REL_BE/.env" ]; then
  diff -u <(grep -v PASSWORD $PROD_BE/.env | sort) <(grep -v PASSWORD $REL_BE/.env | sort) | head -30 | tee -a $REPORT
else
  echo ".env ausente em algum ambiente" | tee -a $REPORT
fi
echo "" | tee -a $REPORT

  SIST="$ENV_PATH/middlewares/Sistema.js"

    echo "--- $(basename $(dirname $(dirname $ENV_PATH))) ---" | tee -a $REPORT
    grep -A 20 "HISTORICO:" $SIST | head -25 | tee -a $REPORT
  fi
done
echo "" | tee -a $REPORT
