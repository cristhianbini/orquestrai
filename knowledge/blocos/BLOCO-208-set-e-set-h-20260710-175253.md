# BLOCO-208 - set +e; set -H

## Script
~~~bash
set +e; set -H
cd /var/www/orquestrai

echo "=== BUSCA EM KNOWLEDGE ==="
grep -r "cafe-real\|cafe real\|cafereal" knowledge/ 2>/dev/null | head -n 10
echo ""

echo "=== ESTRUTURA DE PROJETOS ==="
if [ -d "/var/www/projects" ]; then
    ls -la /var/www/projects/ | grep -i cafe
fi
echo ""

echo "=== BUSCA NO CÓDIGO ==="
find . -type f \( -name "*.js" -o -name "*.mjs" -o -name "*.json" -o -name "*.md" \) -exec grep -l "cafe-real\|cafe real" {} \; 2>/dev/null | head -n 10
echo ""

echo "=== METAS E PENDÊNCIAS ==="
ls -la knowledge/metas/ 2>/dev/null | grep -i cafe
ls -la knowledge/pendencias/ 2>/dev/null | grep -i cafe
~~~
