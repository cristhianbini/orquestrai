# BLOCO-147 - set +e; set -H

## Script
~~~bash
set +e; set -H
DEST_DIR="/var/www/orquestrai/data/knowledge/blocos"
SOURCE_DIR="/var/www/orquestrai"
mkdir -p "$DEST_DIR"
if compgen -G "${SOURCE_DIR}/BLOCO*" > /dev/null; then
  mv -v "${SOURCE_DIR}"/BLOCO* "$DEST_DIR/"
  echo "Movido para $DEST_DIR"
else
  echo "Nenhum arquivo BLOCO* encontrado em $SOURCE_DIR"
fi
echo "--- Conteudo de blocos/ (ultimos 5) ---"
ls -lt "$DEST_DIR" | head -6
~~~
