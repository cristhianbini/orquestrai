# BLOCO-61 - set +e; set +H

## Script
~~~bash
set +e; set +H
REPORT=/tmp/vps_internet_check_$(date +%s).txt
echo "===== DIAGNÓSTICO DE CONECTIVIDADE DA VPS =====" | tee "$REPORT"
echo "" | tee -a "$REPORT"

echo "[1] INTERFACES E ROTAS ATIVAS:" | tee -a "$REPORT"
ip addr show | tee -a "$REPORT"
echo "" | tee -a "$REPORT"
ip route show | tee -a "$REPORT"
echo "" | tee -a "$REPORT"

echo "[2] GATEWAY PADRÃO:" | tee -a "$REPORT"
ip route show default | tee -a "$REPORT"
echo "" | tee -a "$REPORT"

echo "[3] CONFIGURAÇÃO DNS:" | tee -a "$REPORT"
cat /etc/resolv.conf | tee -a "$REPORT"
echo "" | tee -a "$REPORT"

echo "[4] TESTE DE PING PARA IP EXTERNO (8.8.8.8):" | tee -a "$REPORT"
ping -c 3 -W 2 8.8.8.8 | tee -a "$REPORT"
PING_STATUS=$?
echo "Status do ping: $PING_STATUS" | tee -a "$REPORT"
echo "" | tee -a "$REPORT"

echo "[5] TESTE DE CONECTIVIDADE TCP PORTA 53 (8.8.8.8):" | tee -a "$REPORT"
timeout 5 nc -zv 8.8.8.8 53 2>&1 | tee -a "$REPORT"
NC_STATUS=$?
echo "Status do nc: $NC_STATUS" | tee -a "$REPORT"
echo "" | tee -a "$REPORT"

echo "[6] RESOLUÇÃO DNS (google.com):" | tee -a "$REPORT"
dig +short google.com 2>&1 | tee -a "$REPORT"
DIG_STATUS=$?
echo "Status do dig: $DIG_STATUS" | tee -a "$REPORT"
echo "" | tee -a "$REPORT"

echo "[7] NSLOOKUP (google.com):" | tee -a "$REPORT"
nslookup google.com 2>&1 | tee -a "$REPORT"
echo "" | tee -a "$REPORT"

echo "[8] CURL TESTE HTTP (http://google.com):" | tee -a "$REPORT"
timeout 10 curl -I http://google.com 2>&1 | head -5 | tee -a "$REPORT"
CURL_STATUS=$?
echo "Status do curl: $CURL_STATUS" | tee -a "$REPORT"
echo "" | tee -a "$REPORT"

echo "===== RESUMO =====" | tee -a "$REPORT"
[ $PING_STATUS -eq 0 ] && echo "✓ PING: OK" | tee -a "$REPORT" || echo "✗ PING: FALHOU" | tee -a "$REPORT"
[ $NC_STATUS -eq 0 ] && echo "✓ TCP 53: OK" | tee -a "$REPORT" || echo "✗ TCP 53: FALHOU" | tee -a "$REPORT"
[ $DIG_STATUS -eq 0 ] && echo "✓ DNS: OK" | tee -a "$REPORT" || echo "✗ DNS: FALHOU" | tee -a "$REPORT"
[ $CURL_STATUS -eq 0 ] && echo "✓ HTTP: OK" | tee -a "$REPORT" || echo "✗ HTTP: FALHOU" | tee -a "$REPORT"
echo "" | tee -a "$REPORT"
echo "Relatório salvo em: $REPORT" | tee -a "$REPORT"
echo "===== fim BLOCO-INTERNET-CHECK ====="
~~~
