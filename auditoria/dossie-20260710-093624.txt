# S3/CTXNGINXQS01 (2026-07-07): formato de log SEM query string.
# Usado nos locations onde o JWT viaja na URL (?_t=) -- term/ws e
# mas/events|run. Loga $uri (path puro) no lugar de $request (que
# inclui a query completa e vazava o token pro access.log; achado A7).
log_format noquery '$remote_addr - $remote_user [$time_local] "$request_method $uri" $status $body_bytes_sent "$http_referer" "$http_user_agent"';

# ARQUITETURA DESTE PROXY (proxy.conf) -- leia antes de editar
# =====================================================================
# Dois server{} para o mesmo dominio:
#   * server listen 80  (HTTP)  -> so ACME (certbot) + redirect 301 p/ HTTPS.
#                                  Nao serve nada real; tudo vai pro 443.
#   * server listen 443 (HTTPS) -> onde TUDO acontece (app, api, preview).
#
# UPSTREAMS:
#   * web  = orquestrai-web:80    (nginx que serve o SPA/dashboard)
#   * api  = orquestrai-api:3000  (Express: auth, mas, projects, etc)
#   * oqterm = 172.18.0.1:7654    (PTY daemon, fora do Docker)
#
# PREVIEW DE PROJETOS (/projects/{slug}/site/) -- 3 locations no bloco 443,
# nesta ordem (regex e' avaliada na ordem do arquivo, primeira que casa vence):
#   1. location ~ ^/projects/([a-z0-9-]+)/site/  -> serve o site estatico do
#      projeto, protegido por auth_request. Publico: libera; privado: exige
#      ?_t=<jwt>. Usa 'root' + try_files (NAO 'alias' -- ver comentario la).
#   2. location = /internal/preview-auth-443     -> subrequisicao interna de
#      auth (match EXATO, nunca perde pro catch-all). Recebe slug/token por
#      HEADER da location 1 e pergunta pra API se libera.
#   3. location ~ ^/projects/([a-z0-9-]+)/        -> tudo que nao e' /site/
#      (project.json, docs/) -> 403. Impede vazamento de dados do projeto.
#
# Historico do fix (sessao 2026-07-09/10): ver knowledge/licoes/ e os CTXIDs
# CTXPREVIEWMOUNT01, CTXPREVIEWAUTHFIX02, CTXPREVIEWALIASFIX01, CTXPREVIEWBLOCK403.
# =====================================================================

        access_log /var/log/nginx/access.log noquery;
        limit_req zone=oq_term burst=15 nodelay;  # B28_TERM_RL
        proxy_pass http://172.18.0.1:7654;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;

        proxy_buffering off;
        proxy_read_timeout 24h;
        proxy_send_timeout 24h;
    }
    # <<< OQ-67a21 WS canonical <<<

    # [CTXPREVIEWAUTH03] LOCATION 1: serve o site estatico do projeto.
    # Protegida por auth_request (subreq p/ a API decidir publico/privado).
    # set $preview_slug/$preview_token: capturam slug e ?_t= AQUI para
    # repassar por HEADER a location interna -- NAO usar $1 cru na diretiva
    # auth_request (isso roteava a subreq pro catch-all web:80 em vez da
    # location de auth; foi a causa raiz do bug de 2026-07-09, ver licao).
    location ~ ^/projects/([a-z0-9-]+)/site/ {
        set $preview_slug $1;
        set $preview_token $arg__t;
        auth_request /internal/preview-auth-443;
        access_log /var/log/nginx/access.log noquery;
        # ROOT, nao ALIAS: 'alias' + location regex + try_files e' antipadrao
        # do nginx -- com alias, $uri no try_files vira o path completo da
        # request, gerando caminho duplicado e 404 mesmo com o arquivo certo
        # no disco. Com 'root', o path da URL e' anexado ao root e bate com
        # o layout real (/var/www/orquestrai + /projects/{slug}/site/...).
        root /var/www/orquestrai;
        add_header Content-Security-Policy "default-src 'self' 'unsafe-inline'; img-src 'self' data:; connect-src 'none'; frame-ancestors 'self'" always;
        add_header X-Frame-Options "SAMEORIGIN" always;
        try_files $uri $uri/ /projects/$preview_slug/site/index.html =404;
    }
    # [CTXPREVIEWBLOCK403] tudo sob /projects/ que NAO seja /site/ -> 403
    # (project.json, docs/, etc). A location /site/ vem antes e vence por
    # ordem de regex nos paths com /site/; o resto cai aqui.
    location ~ ^/projects/([a-z0-9-]+)/ {
        return 403;
    }

    # [CTXPREVIEWAUTH03] LOCATION 2: subrequisicao interna de auth, chamada
    # pelo auth_request da LOCATION 1. 'internal' = inacessivel de fora.
    # location = (MATCH EXATO): de proposito -- tem a maior precedencia do
    # nginx, entao NUNCA perde a subreq pro catch-all 'location /' (era o
    # que acontecia com regex+$1: a subreq caia no web:80 e o auth sempre
    # "passava", liberando preview privado sem token). Recebe slug e token
    # por HEADER (X-Preview-Slug/X-Preview-Token, setados na LOCATION 1) e
    # repassa pra API, que responde 200 (libera) ou 401 (nega). Sufixo -443
    # mantem o nome unico vs. o bloco 80 (nomes internos sao por server{}).
    location = /internal/preview-auth-443 {
        internal;
        access_log off;
        proxy_pass http://api:3000/api/projects/$preview_slug/preview-auth;
        proxy_pass_request_body off;
        proxy_set_header Content-Length "";
        proxy_set_header X-Preview-Slug $preview_slug;
        proxy_set_header X-Preview-Token $preview_token;
    }

    # B28_AUTH_RL — rate-limit para endpoints de autenticacao
    location ^~ /api/auth/ {
        limit_req zone=oq_auth burst=20 nodelay;
        set $upstream_api http://orquestrai-api:3000;
        proxy_pass $upstream_api;
        proxy_http_version 1.1;

        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_read_timeout 30s;
    }

    # CTXSSE01: SSE do MAS precisa de buffer off + timeout longo (runs de 1-2 min)

        # S3/CTXNGINXQS01: token JWT viaja em ?_t= aqui -- log sem query
        access_log /var/log/nginx/access.log noquery;
        set $upstream_api http://orquestrai-api:3000;
