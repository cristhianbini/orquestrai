# BLOCO-218 - set +e; set -H

## Script
~~~bash
set +e; set -H

echo "=== BACKUP CLUSTER.DB - PROCESSO COMPLETO ==="
echo ""
echo "Passo 1: Definir caminhos e timestamp"
DB_PATH="/var/www/orquestrai/data/cluster.db"
BACKUP_DIR="/var/www/orquestrai/data/backups"
TIMESTAMP=$(date +%Y%m%d-%H%M%S)
BACKUP_FILE="cluster.db.bak-$TIMESTAMP"

echo "  DB original: $DB_PATH"
echo "  Diretório backup: $BACKUP_DIR"
echo "  Arquivo backup: $BACKUP_FILE"
echo ""

echo "Passo 2: Verificar se DB existe"
if [ -f "$DB_PATH" ]; then
    echo "  ✓ Arquivo encontrado"
    ls -lh "$DB_PATH"
else
    echo "  ✗ Arquivo NÃO encontrado. Abortando."
    exit 1
fi
echo ""

echo "Passo 3: Criar diretório de backups se não existir"
mkdir -p "$BACKUP_DIR"
echo "  ✓ Diretório garantido em: $BACKUP_DIR"
echo ""

echo "Passo 4: Copiar arquivo com preservação de atributos"
cp -a "$DB_PATH" "$BACKUP_DIR/$BACKUP_FILE"
COPY_EXIT=$?
if [ $COPY_EXIT -eq 0 ]; then
    echo "  ✓ Backup copiado com sucesso"
else
    echo "  ✗ Erro ao copiar (exit code: $COPY_EXIT)"
    exit 1
fi
echo ""

echo "Passo 5: Validar integridade do backup"
sha256sum "$DB_PATH" > "$BACKUP_DIR/$BACKUP_FILE.sha256"
sha256sum -c "$BACKUP_DIR/$BACKUP_FILE.sha256" > /dev/null 2>&1
VALIDATE_EXIT=$?
if [ $VALIDATE_EXIT -eq 0 ]; then
    echo "  ✓ Checksum validado"
else
    echo "  ✗ Checksum falhou"
    exit 1
fi
echo ""

echo "Passo 6: Listar últimos 3 backups"
ls -lh "$BACKUP_DIR"/cluster.db.bak-* | tail -n 3
echo ""

echo "=== RESUMO FINAL ==="
echo "Backup salvo em: $BACKUP_DIR/$BACKUP_FILE"
echo "Tamanho: $(ls -lh $BACKUP_DIR/$BACKUP_FILE | awk '{print $5}')"
echo "Checksum registrado em: $BACKUP_DIR/$BACKUP_FILE.sha256"
~~~
