# BLOCO-79 - set +e; set -H

## Script
~~~bash
set +e; set -H
echo "[INFO] Listando conteúdo da base de conhecimento em /var/www/orquestrai/knowledge..."
ls -lh /var/www/orquestrai/knowledge/ 2>/dev/null || echo "[ERRO] Diretorio base knowledge nao encontrado."
echo "[INFO] Listando contextos..."
ls -lh /var/www/orquestrai/knowledge/contextos/ 2>/dev/null || echo "[AVISO] Diretorio contextos nao encontrado."
echo "[INFO] Exibindo changelog (se existir)..."
if [ -f /var/www/orquestrai/knowledge/changelog.md ]; then
  tail -n 15 /var/www/orquestrai/knowledge/changelog.md
else
  echo "[AVISO] Arquivo changelog.md inexistente."
fi
~~~
