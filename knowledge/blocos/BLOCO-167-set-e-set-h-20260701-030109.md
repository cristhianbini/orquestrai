# BLOCO-167 - set +e; set -H

## Script
~~~bash
set +e; set -H
BASE="/var/www/orquestrai/knowledge"
if [ -f "$BASE/changelog.md" ]; then
    echo "=== Changelog Recente ==="
    tail -n 15 "$BASE/changelog.md"
else
    echo "Changelog não encontrado em $BASE."
fi
echo ""
if [ -d "$BASE/pendencias" ]; then
    echo "=== Últimas Pendências ==="
    ls -lt "$BASE/pendencias/" | head -n 5
fi
echo ""
if [ -d "$BASE/blocos" ]; then
    echo "=== Última Auditoria LAVE ==="
    ls -lt "$BASE/blocos/" | head -n 5
fi
~~~
