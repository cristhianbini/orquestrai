# BLOCO-62 - set +e; set +H

## Script
~~~bash
set +e; set +H
REPORT=/tmp/check_internet_$(date +%s).txt
echo "===== VERIFICACAO DE CONECTIVIDADE - VPS CBINI =====" | tee $REPORT
echo "" | tee -a $REPORT

echo "[1] Configuracao de rede (interfaces e IPs)" | tee -a $REPORT
ip addr show | tee -a $REPORT
echo "" | tee -a $REPORT

echo "[2] Tabela de rotas" | tee -a $REPORT
ip route show | tee -a $REPORT
echo "" | tee -a $REPORT

echo "[3] DNS configurado (/etc/resolv.conf)" | tee -a $REPORT
cat /etc/resolv.conf | tee -a $REPORT
echo "" | tee -a $REPORT

echo "[4] Teste de ping para Google DNS (8.8.8.8)" | tee -a $REPORT
ping -c 4 8.8.8.8 2>&1 | tee -a $REPORT
echo "" | tee -a $REPORT

echo "[5] Teste de ping para dominio externo (google.com)" | tee -a $REPORT
ping -c 4 google.com 2>&1 | tee -a $REPORT
echo "" | tee -a $REPORT

echo "[6] Teste de resolucao DNS (nslookup)" | tee -a $REPORT
nslookup google.com 2>&1 | tee -a $REPORT
echo "" | tee -a $REPORT

echo "[7] Teste HTTP externo (curl google.com)" | tee -a $REPORT
curl -I --max-time 10 http://google.com 2>&1 | head -n 10 | tee -a $REPORT
echo "" | tee -a $REPORT

echo "[8] Regras de firewall iptables" | tee -a $REPORT
sudo iptables -L -n -v | tee -a $REPORT
echo "" | tee -a $REPORT

echo "[9] Status de servicos de rede" | tee -a $REPORT
systemctl status networking 2>&1 | head -n 20 | tee -a $REPORT
echo "" | tee -a $REPORT

echo "RELATORIO COMPLETO: $REPORT"
echo "===== fim BLOCO-INTERNET-CHECK ====="
~~~
