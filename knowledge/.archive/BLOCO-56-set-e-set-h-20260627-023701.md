set +e; set +H

PROD_BE=/var/www/xmonex/backend
TESTE_BE=/var/www/xmonex_teste/backend
RELEASE_BE=/var/www/xmonex_release/backend

echo "===== BLOCO-UPLOAD-CONFIG-CHECK =====" | tee $REPORT
echo "Data: $(date)" | tee -a $REPORT
echo "" | tee -a $REPORT

echo "[1] VERIFICAÇÃO APP.JS/SERVER.JS - bodyParser/express limits" | tee -a $REPORT
for APP_FILE in $PROD_BE/app.js $PROD_BE/server.js $TESTE_BE/app.js $TESTE_BE/server.js $RELEASE_BE/app.js $RELEASE_BE/server.js; do
    if [ -f "$APP_FILE" ]; then
        echo ">>> $APP_FILE" | tee -a $REPORT
        grep -n -A 2 -B 2 -E "(bodyParser|express\.(json|urlencoded)).*limit" "$APP_FILE" 2>/dev/null | tee -a $REPORT
    fi
done
echo "" | tee -a $REPORT

echo "[2] VERIFICAÇÃO MIDDLEWARES - multer/formidable/busboy limits" | tee -a $REPORT
for MW_DIR in $PROD_BE/middlewares $PROD_BE/routes $TESTE_BE/middlewares $TESTE_BE/routes $RELEASE_BE/middlewares $RELEASE_BE/routes; do
    if [ -d "$MW_DIR" ]; then
        echo ">>> Buscando em $MW_DIR" | tee -a $REPORT
        find "$MW_DIR" -type f -name "*.js" -exec grep -l -E "(multer|formidable|busboy|express-fileupload)" {} \; 2>/dev/null | while read MWFILE; do
            echo "  -> $MWFILE" | tee -a $REPORT
            grep -n -A 3 -B 1 -E "(multer|formidable|busboy|fileupload).*\(" "$MWFILE" 2>/dev/null | grep -E "(limit|fileSize|maxFileSize)" | tee -a $REPORT
        done
    fi
done
echo "" | tee -a $REPORT

echo "[3] VERIFICAÇÃO CONFIG/* - arquivos de configuração" | tee -a $REPORT
for CFG_DIR in $PROD_BE/config $TESTE_BE/config $RELEASE_BE/config; do
    if [ -d "$CFG_DIR" ]; then
        echo ">>> $CFG_DIR" | tee -a $REPORT
        find "$CFG_DIR" -type f \( -name "*.js" -o -name "*.json" \) -exec grep -l -E "(limit|maxFileSize|fileSize)" {} \; 2>/dev/null | while read CFGFILE; do
            echo "  -> $CFGFILE" | tee -a $REPORT
            grep -n -E "(limit|maxFileSize|fileSize)" "$CFGFILE" 2>/dev/null | tee -a $REPORT
        done
    fi
done
echo "" | tee -a $REPORT

echo "[4] NGINX - conf.d e includes auxiliares" | tee -a $REPORT
if [ -d /etc/nginx/conf.d ]; then
    echo ">>> /etc/nginx/conf.d/" | tee -a $REPORT
    ls -lh /etc/nginx/conf.d/ 2>/dev/null | tee -a $REPORT
    for CONF_FILE in /etc/nginx/conf.d/*.conf; do
        if [ -f "$CONF_FILE" ]; then
            echo "  -> $CONF_FILE" | tee -a $REPORT
            grep -n -E "(client_max_body_size|client_body_buffer_size|proxy_buffers|proxy_busy_buffers_size)" "$CONF_FILE" 2>/dev/null | tee -a $REPORT
        fi
    done
fi
echo "" | tee -a $REPORT

echo "[5] NGINX - sites-enabled (symlinks ativos)" | tee -a $REPORT
ls -lh /etc/nginx/sites-enabled/ 2>/dev/null | tee -a $REPORT
for SITE_ENABLED in /etc/nginx/sites-enabled/*; do
    if [ -f "$SITE_ENABLED" ] || [ -L "$SITE_ENABLED" ]; then
        echo ">>> $SITE_ENABLED" | tee -a $REPORT
        grep -n -E "(client_max_body_size|client_body_buffer_size|proxy_request_buffering)" "$SITE_ENABLED" 2>/dev/null | tee -a $REPORT
    fi
done
echo "" | tee -a $REPORT
