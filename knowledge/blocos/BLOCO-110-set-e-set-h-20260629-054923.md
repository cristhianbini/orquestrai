# BLOCO-110 - set +e; set +H

## Script
~~~bash
set +e; set +H

echo "=== STATUS DO ORQUESTRAI ==="
cd /var/www/orquestrai 2>/dev/null && pwd || echo "Diretorio orquestrai nao encontrado"

echo -e "\n=== CONTAINERS DOCKER ==="
docker ps --format "table {{.Names}}\t{{.Status}}\t{{.Ports}}"

echo -e "\n=== ULTIMAS LINHAS DE LOG ==="
docker logs orquestrai-api --tail 20 2>&1 || echo "Container api nao encontrado"

echo -e "\n=== ESPACO EM DISCO ==="
df -h / | tail -1

echo -e "\n=== ULTIMAS PENDENCIAS ==="
ls -lt /var/www/orquestrai/knowledge/pendencias/ 2>/dev/null | head -5 || echo "Sem pendencias registradas"
~~~
