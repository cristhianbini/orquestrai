# BLOCO-144 - set +e; set +H

## Script
~~~bash
set +e; set +H
echo "### VERIFICACAO KNOWLEDGE BASE ###"
BASE="/var/www/orquestrai/knowledge"
DATA_BASE="/var/www/orquestrai/data/knowledge/blocos"

if [ -d "$BASE" ]; then
    echo "[Estrutura] $BASE"
    ls -la "$BASE"
    echo ""
    for dir in metas pendencias licoes decisoes contextos blocos; do
        if [ -d "$BASE/$dir" ]; then
            echo "[$dir]"
            ls "$BASE/$dir"
        else
            echo "[$dir] NAO EXISTE"
        fi
        echo ""
    done
    if [ -f "$BASE/changelog.md" ]; then
        echo "[changelog.md - ultimas 5 linhas]"
        tail -n 5 "$BASE/changelog.md"
    fi
else
    echo "ERRO: $BASE nao encontrado."
fi

echo "### VERIFICACAO AUDITORIA LAVE ###"
if [ -d "$DATA_BASE" ]; then
    echo "[$DATA_BASE]"
    ls -la "$DATA_BASE"
else
    echo "Diretorio de auditoria $DATA_BASE nao encontrado."
fi
~~~
