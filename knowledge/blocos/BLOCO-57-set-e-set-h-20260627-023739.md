# BLOCO-57 - set +e; set +H

## Script
~~~bash
set +e; set +H
REPORT=/tmp/nginx_includes_audit_$(date +%s).txt
echo "===== BLOCO-NGINX-INCLUDES-AUDIT =====" | tee $REPORT
echo "[1] Verificando includes declarados em /etc/nginx/nginx.conf" | tee -a $REPORT
grep -n "include" /etc/nginx/nginx.conf 2>/dev/null | tee -a $REPORT
echo "" | tee -a $REPORT

echo "[2] Conteúdo de /etc/nginx/sites-available/xmonex.cbini.com.br" | tee -a $REPORT
if [ -f /etc/nginx/sites-available/xmonex.cbini.com.br ]; then
    cat -n /etc/nginx/sites-available/xmonex.cbini.com.br | tee -a $REPORT
    echo "" | tee -a $REPORT
    echo "  -> Buscando diretivas de buffer/proxy:" | tee -a $REPORT
    grep -n -E "(client_max_body_size|client_body_buffer_size|proxy_request_buffering|proxy_buffers|proxy_busy_buffers_size|proxy_buffer_size)" /etc/nginx/sites-available/xmonex.cbini.com.br 2>/dev/null | tee -a $REPORT
else
    echo "  AVISO: Arquivo não encontrado!" | tee -a $REPORT
fi
echo "" | tee -a $REPORT

echo "[3] Arquivos .include referenciados em sites-available" | tee -a $REPORT
INCLUDES=$(grep -h "include" /etc/nginx/sites-available/xmonex.cbini.com.br 2>/dev/null | grep -v "^#" | awk '{print $2}' | tr -d ';')
for INC in $INCLUDES; do
    if [ -f "$INC" ]; then
        echo "  -> Analisando: $INC" | tee -a $REPORT
        cat -n "$INC" | tee -a $REPORT
        echo "" | tee -a $REPORT
    elif [ -d "$INC" ]; then
        echo "  -> Diretório include: $INC" | tee -a $REPORT
        ls -lh "$INC" | tee -a $REPORT
    fi
done
echo "" | tee -a $REPORT

echo "[4] Busca recursiva por .conf e .include em /etc/nginx/" | tee -a $REPORT
find /etc/nginx/ -type f \( -name "*.conf" -o -name "*.include" \) -exec echo ">>> {}" \; -exec grep -Hn -E "(client_max_body_size|proxy_request_buffering|proxy_buffers)" {} \; 2>/dev/null | tee -a $REPORT
echo "" | tee -a $REPORT

echo "[5] Teste de sintaxe nginx (nginx -t)" | tee -a $REPORT
nginx -t 2>&1 | tee -a $REPORT
echo "" | tee -a $REPORT

echo "Relatório salvo em: $REPORT" | tee -a $REPORT
echo "===== FIM BLOCO-NGINX-INCLUDES-AUDIT ====="
~~~
