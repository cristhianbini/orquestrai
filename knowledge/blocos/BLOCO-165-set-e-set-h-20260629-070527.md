# BLOCO-165 - set +e; set +H

## Script
~~~bash
set +e; set +H
LOG_FILE="/var/www/orquestrai/knowledge/changelog.md"
CTX_DIR="/var/www/orquestrai/knowledge/contextos"

if [ -d "/var/www/orquestrai/knowledge" ]; then
    if [ -f "$LOG_FILE" ]; then
        echo "=== ULTIMAS ENTRADAS NO CHANGELOG ==="
        tail -n 10 "$LOG_FILE"
        echo ""
    else
        echo "AVISO: changelog.md nao encontrado em $LOG_FILE."
    fi
    echo "=== ARQUIVOS DE CONTEXTO (Modificacao Recente) ==="
    ls -lht "$CTX_DIR" 2>/dev/null | head -8 || echo "Diretorio de contextos inacessivel ou vazio."
else
    echo "ERRO CRITICO: O diretorio de conhecimento /var/www/orquestrai/knowledge/ nao existe."
fi
~~~
